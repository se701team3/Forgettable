import React, {useState} from 'react';
import logo from './logo.svg';
import './App.css';
import Modal from './components/Dialog/Dialog.js';
import {fileURLToPath} from 'url';

function App() {
  const [isDialogShow, setDialogShow] = useState(false);
  return (
    <div className='App'>
      <header className='App-header'>
        <img src={logo} className='App-logo' alt='logo' />
        <p>
          Edit <code>src/App.tsx</code> and save to reload.
        </p>
        <a
          className='App-link'
          href='https://github.com/se701team3/Forgettable.git'
          target='_blank'
          rel='noopener noreferrer'
        >
          Hello
        </a>
        <button
          className='App-link'
          onClick={() => {
            setDialogShow(!isDialogShow);
          }}
        >
          Open Modal
        </button>
        {isDialogShow ? (
          <Modal
            type={'newEntry'}
            title={'Warning'}
            description={'Are you sure you want to delete this person? You cannot undo this action.'}
            input1={'Platform:'}
            hint1={'Insert a platform'}
            input2={'URL Link:'}
            hint2={'Insert a url link'}
            cancalEvent={() => {
              setDialogShow(!isDialogShow);
            }}
            confirmEvent={() => {
              setDialogShow(!isDialogShow);
            }}
          />
        ) : null}
      </header>
    </div>
  );
}

export default App;
